3.  DATABASE SCHEMA (PostgreSQL)
────────────────────────────────────────────────────────────────────────────────

─── 3a. Users ──────────────────────────────────────────────────────────────────

  TABLE users
  ┌──────────────────┬────────────────────────────┬──────────────────────────────┐
  │ Column           │ Type                       │ Notes                       │
  ├──────────────────┼────────────────────────────┼──────────────────────────────┤
  │ id               │ UUID (PK)                  │                             │
  │ email            │ VARCHAR(255) UNIQUE NOT NULL│                             │
  │ password_hash    │ VARCHAR(255) NOT NULL       │ Werkzeug generated          │
  │ display_name     │ VARCHAR(100)                │                             │
  │ tier             │ VARCHAR(20) DEFAULT 'core'  │ 'core' | 'pro' | 'admin'   │
  │ is_active        │ BOOLEAN DEFAULT TRUE        │                             │
  │ onboarding_complete │ BOOLEAN DEFAULT FALSE    │ Used by future onboarding   │
  │ avatar_url      │ VARCHAR(500)                │ FUTURE                      │
  │ created_at       │ TIMESTAMP WITH TZ           │ DEFAULT NOW()               │
  │ updated_at       │ TIMESTAMP WITH TZ           │ AUTO-UPDATE                 │
  │ last_login_at    │ TIMESTAMP WITH TZ           │                             │
  └──────────────────┴────────────────────────────┴──────────────────────────────┘

  Relationships: progress (→ UserProgress), ai_interactions (→ AiInteraction)

─── 3b. Sessions (audit trail) ─────────────────────────────────────────────────

  TABLE sessions
  ┌──────────────────┬────────────────────────────┬──────────────────────────────┐
  │ Column           │ Type                       │ Notes                       │
  ├──────────────────┼────────────────────────────┼──────────────────────────────┤
  │ id               │ UUID (PK)                  │                             │
  │ user_id          │ UUID (FK → users.id)       │ ON DELETE CASCADE           │
  │ token            │ VARCHAR(255) UNIQUE         │ Session token               │
  │ ip_address       │ VARCHAR(45)                 │                             │
  │ user_agent       │ TEXT                        │                             │
  │ expires_at       │ TIMESTAMP WITH TZ           │                             │
  │ created_at       │ TIMESTAMP WITH TZ           │                             │
  └──────────────────┴────────────────────────────┴──────────────────────────────┘

─── 3c. Modules (curriculum grouping) ──────────────────────────────────────────

  TABLE modules
  ┌──────────────────┬────────────────────────────┬──────────────────────────────┐
  │ Column           │ Type                       │ Notes                       │
  ├──────────────────┼────────────────────────────┼──────────────────────────────┤
  │ id               │ UUID (PK)                  │                             │
  │ title            │ VARCHAR(200) NOT NULL       │                             │
  │ slug             │ VARCHAR(200) UNIQUE NOT NULL│ e.g. "what-is-the-market"   │
  │ description      │ TEXT                        │                             │
  │ icon             │ VARCHAR(50)                 │ SVG icon name               │
  │ sort_order       │ INTEGER DEFAULT 0           │ Display order               │
  │ is_published     │ BOOLEAN DEFAULT FALSE       │                             │
  │ created_at       │ TIMESTAMP WITH TZ           │                             │
  │ updated_at       │ TIMESTAMP WITH TZ           │                             │
  └──────────────────┴────────────────────────────┴──────────────────────────────┘

  Relationships: lessons (→ Lesson, cascade delete)

─── 3d. Lessons ────────────────────────────────────────────────────────────────

  TABLE lessons
  ┌──────────────────┬────────────────────────────┬──────────────────────────────┐
  │ Column           │ Type                       │ Notes                       │
  ├──────────────────┼────────────────────────────┼──────────────────────────────┤
  │ id               │ UUID (PK)                  │                             │
  │ module_id        │ UUID (FK → modules.id)     │ ON DELETE CASCADE           │
  │ title            │ VARCHAR(200) NOT NULL       │                             │
  │ slug             │ VARCHAR(200) NOT NULL       │ Unique per module           │
  │ content          │ TEXT                        │ HTML content                │
  │ content_type     │ VARCHAR(20) DEFAULT 'html'  │ 'html' | 'markdown'        │
  │ sort_order       │ INTEGER DEFAULT 0           │                             │
  │ estimated_minutes│ INTEGER DEFAULT 5           │                             │
  │ is_published     │ BOOLEAN DEFAULT FALSE       │                             │
  │ concept_tags     │ TEXT[]                      │ Array e.g. {"liquidity"}    │
  │ related_signal_type│ VARCHAR(50)               │ Links to bot signal type    │
  │ created_at       │ TIMESTAMP WITH TZ           │                             │
  │ updated_at       │ TIMESTAMP WITH TZ           │                             │
  └──────────────────┴────────────────────────────┴──────────────────────────────┘

  UNIQUE(module_id, slug)

─── 3e. User Progress ─────────────────────────────────────────────────────────

  TABLE user_progress
  ┌──────────────────┬────────────────────────────┬──────────────────────────────┐
  │ Column           │ Type                       │ Notes                       │
  ├──────────────────┼────────────────────────────┼──────────────────────────────┤

  │ user_id          │ UUID (FK → users.id)       │ ON DELETE CASCADE           │
  │ lesson_id        │ UUID (FK → lessons.id)     │ ON DELETE CASCADE           │

  │ created_at       │ TIMESTAMP WITH TZ           │                             │
  │ updated_at       │ TIMESTAMP WITH TZ           │                             │
  └──────────────────┴────────────────────────────┴──────────────────────────────┘

  UNIQUE(user_id, lesson_id)

─── 3f. Bot Events ─────────────────────────────────────────────────────────────

  TABLE bot_events
  ┌──────────────────┬────────────────────────────┬──────────────────────────────┐
  │ Column           │ Type                       │ Notes                       │
  ├──────────────────┼────────────────────────────┼──────────────────────────────┤
  │ id               │ BIGSERIAL (PK)             │ Auto-increment (log-style)  │
  │ timestamp        │ TIMESTAMP WITH TZ           │ ET timezone stored          │
  │ event            │ VARCHAR(50) NOT NULL        │ 'scan' | 'entry' | 'exit'  │
  │ symbol           │ VARCHAR(10)                 │                             │
  │ side             │ VARCHAR(10)                 │ 'long' | 'short' | 'none'  │
  │ price            │ DOUBLE PRECISION            │                             │
  │ qty              │ INTEGER                     │                             │
  │ status           │ VARCHAR(30)                 │ 'filled' | 'closed' | etc. │
  │ message          │ TEXT                        │ Free-form context           │
  │ created_at       │ TIMESTAMP WITH TZ           │ DEFAULT NOW()               │
  └──────────────────┴────────────────────────────┴──────────────────────────────┘

  INDEX idx_bot_events_event_ts ON bot_events(event, timestamp)
